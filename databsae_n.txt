create database saidb;
use saidb;
drop table  registration;
drop table site_user;
drop table site_user_info;
drop table upload;
drop table enrolment;
drop table notice;
drop table event;
drop table marks;
drop table attendance;
DROP TABLE user_pass;
create table registration(reg_id INTEGER UNSIGNED  NOT NULL AUTO_INCREMENT,enroll VARCHAR(10),std_name VARCHAR(30),father_name VARCHAR(30),dob DATE,corrspndg_add VARCHAR(255),permnt_add VARCHAR(150),cntct_no TINYBLOB,qualification    VARCHAR(25),email_id VARCHAR(40),draft_no INTEGER UNSIGNED NOT NULL DEFAULT 0,reg_no int(10),date_reg date,amnt float(10,3),duration_inmonth int(10),strt_frm    date,PRIMARY KEY(reg_id)) ENGINE=MYISAM ;
create table site_user(user_id INTEGER NOT NULL AUTO_INCREMENT,user_name VARCHAR(20) NOT NULL unique,email  VARCHAR(40) NOT NULL DEFAULT '',u_pass VARCHAR(50) NOT NULL,admin_level TINYINT UNSIGNED NOT NULL DEFAULT 0 , PRIMARY KEY(user_id));
create table site_user_info(user_id INTEGER NOT NULL AUTO_INCREMENT,user_name varchar(20),first_name VARCHAR(20) NOT NULL,last_name VARCHAR(20)NOT NULL,dob DATE,email VARCHAR(40) NOT NULL,city VARCHAR(20),state CHAR(20),country VARCHAR(20),branch VARCHAR(255),FOREIGN KEY(user_id) REFERENCES site_user(user_id))ENGINE=MYISAM;
CREATE TABLE user_pass(user_id INTEGER NOT NULL AUTO_INCREMENT,user_name VARCHAR(20) NOT NULL,u_pass VARCHAR(20) NOT NULL,FOREIGN KEY(user_id) REFERENCES site_user(user_id))ENGINE=MYISAM;
create table upload(id int(10) not null auto_increment primary key,file varchar(200) not null,type varchar(200),size int(255))engine=myisam;
create table enrolment(id int(10) not null auto_increment primary key,enr varchar(10));
create table notice(n_id int(10) not null auto_increment primary key,notice varchar(500));
create table event(e_id int(10) not null auto_increment primary key,event varchar(500));
create table marks(m_id int(10) not null auto_increment primary key,marks varchar(500));
create table attendance(a_id int(10) not null auto_increment primary key,attendance varchar(500));
